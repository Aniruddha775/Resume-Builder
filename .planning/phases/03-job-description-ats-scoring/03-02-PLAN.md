3. Create `src/lib/scoring/fuse-matcher.ts`:

   ```typescript
   import Fuse from 'fuse.js'
   import keywordAliases from '@/data/keyword-aliases.json'

   // The aliases file maps canonical term -> aliases array
   // E.g. { "JavaScript": ["JS", "Javascript", "ECMAScript"] }
   type AliasMap = Record<string, string[]>

   interface CorpusEntry {
     canonical: string
     term: string
   }

   // Build a Fuse corpus from a list of canonical keyword terms.
   // For each canonical term, adds the term itself plus all its aliases as separate entries
   // pointing back to the canonical. This lets us search resume words against the corpus
   // and know which canonical keyword matched.
   function buildCorpus(canonicalTerms: string[]): CorpusEntry[] {
     const aliasMap = keywordAliases as AliasMap
     const entries: CorpusEntry[] = []
     for (const canonical of canonicalTerms) {
       entries.push({ canonical, term: canonical })
       const aliases = aliasMap[canonical] ?? []
       for (const alias of aliases) {
         entries.push({ canonical, term: alias })
       }
     }
     return entries
   }

   /**
    * For each keyword in `keywords`, returns true if that keyword (or any of its aliases)
    * appears in `resumeText` via fuzzy matching.
    *
    * CRITICAL: We search resume tokens against the keyword corpus — NOT the reverse.
    * Fuse.js searches a collection for a query. Our collection = keyword + alias entries.
    * Our queries = resume text tokens (words). We find which canonicals matched.
    *
    * threshold=0.3: tight enough to match abbreviations, strict enough to avoid false positives.
    * ignoreLocation=true: keywords can appear anywhere in the resume, not just position 0.
    */
   export function matchKeywords(keywords: string[], resumeText: string): boolean[] {
     if (keywords.length === 0) return []
     if (!resumeText.trim()) return keywords.map(() => false)

     const corpus = buildCorpus(keywords)

     const fuse = new Fuse(corpus, {
       keys: ['term'],
       threshold: 0.3,
       ignoreLocation: true,
       minMatchCharLength: 2,
       includeScore: false,
     })

     // Tokenize resume text: split on whitespace and punctuation, filter short tokens
     const resumeTokens = resumeText
       .split(/[\s,;:|/\\()\[\]{}<>'"]+/)
       .filter((w) => w.length >= 2)

     const matchedCanonicals = new Set<string>()

     for (const token of resumeTokens) {
       const results = fuse.search(token)
       for (const result of results) {
         matchedCanonicals.add(result.item.canonical)
       }
     }

     return keywords.map((kw) => matchedCanonicals.has(kw))
   }
   ```

   import type { Resume } from '@/types/resume'
   import type { ScoreCategoryBreakdown } from '@/types/score'

   // Weight for the formatting component — must match WEIGHTS in ats-scorer.ts
   const FORMATTING_WEIGHT = 0.10

   /**
    * Checks resume structural compliance for ATS friendliness.
    * These are data-level checks — we verify section presence via the resume data model,
    * NOT via DOM inspection. Our PDF renderer already enforces single-column layout.
    *
    * All 5 checks:
    * 1. Contact section: fullName AND email are non-empty
    * 2. Experience section: at least 1 entry
    * 3. Education section: at least 1 entry
    * 4. Skills section: at least 1 skill group
    * 5. Summary: non-empty and at least 20 characters
    */
   export function checkFormatting(resume: Resume): ScoreCategoryBreakdown {
     const s = resume.sections
     const checks: boolean[] = [
       Boolean(s.contactInfo.fullName?.trim() && s.contactInfo.email?.trim()),
       s.experience.length > 0,
       s.education.length > 0,
       s.skills.length > 0,
       Boolean(s.summary && s.summary.trim().length > 20),
     ]

     const passedCount = checks.filter(Boolean).length
     const total = checks.length
     const percentage = Math.round((passedCount / total) * 100)

     return {
       category: 'Formatting',
       weight: FORMATTING_WEIGHT,
       matched: passedCount,
       total,
       percentage,
       weightedScore: FORMATTING_WEIGHT * percentage,
     }
   }

2. Create `src/lib/scoring/ats-scorer.ts`:

   ```typescript
   import type { Resume } from '@/types/resume'
   import type { KeywordSet, Keyword } from '@/types/job-description'
   import type { ScoreResult, ScoreCategoryBreakdown } from '@/types/score'
   import { matchKeywords } from './fuse-matcher'
   import { checkFormatting } from './formatting-checker'

   // AUTHORITATIVE WEIGHTS: from ROADMAP.md SCORE-02 (Phase 3 spec).
   // Note: DESIGN.md §ATS Scoring Algorithm shows different weights (hard=0.35, tools=0.15, formatting=0.20).
   // ROADMAP SCORE-02 supersedes DESIGN.md for this phase. DESIGN.md should be updated post-Phase 3.
   // Sum: 0.40 + 0.20 + 0.20 + 0.10 + 0.10 = 1.00 ✓
   export const WEIGHTS = {
     hardRequirements: 0.40,
     preferredSkills:  0.20,
     toolsAndTech:     0.20,
     softSkills:       0.10,
     formatting:       0.10,
   } as const

   /**
    * Extracts a single flat text string from all resume sections.
    * This is the text searched for keyword matches.
    */
   function extractResumeText(resume: Resume): string {
     const s = resume.sections
     const parts: string[] = []

     // Contact info
     parts.push(s.contactInfo.fullName ?? '')

     // Summary
     if (s.summary) parts.push(s.summary)

     // Experience: company, title, all bullets
     for (const exp of s.experience) {
       parts.push(exp.company, exp.title)
       parts.push(...exp.bullets)
     }

     // Education
     for (const edu of s.education) {
       parts.push(edu.institution, edu.degree, edu.field)
     }

     // Skills: category name + all items
     for (const group of s.skills) {
       parts.push(group.category)
       parts.push(...group.items)
     }

     return parts.filter(Boolean).join(' ')
   }

   /**
    * Scores a single keyword category against the resume text.
    *
    * IMPORTANT — empty category behavior:
    * If keywords.length === 0, this category contributes 0 weighted points.
    * (NOT 100% * weight, which would artificially inflate the score for JDs
    * that don't mention soft skills.)
    */
   function scoreCategory(
     keywords: Keyword[],
     resumeText: string,
     weight: number,
     label: string
   ): ScoreCategoryBreakdown {
     if (keywords.length === 0) {
       return {
         category: label,
         weight,
         matched: 0,
         total: 0,
         percentage: 0,
         weightedScore: 0,
       }
     }

     const terms = keywords.map((k) => k.term)
     const matchResults = matchKeywords(terms, resumeText)
     const matchedCount = matchResults.filter(Boolean).length
     const percentage = Math.round((matchedCount / keywords.length) * 100)

     return {
       category: label,
       weight,
       matched: matchedCount,
       total: keywords.length,
       percentage,
       weightedScore: weight * percentage,
     }
   }

   /**
    * Main ATS scoring function.
    *
    * PURE FUNCTION — no async, no LLM, no DOM access.
    * Runs synchronously in < 5ms for typical resume + keyword sets.
    * Called by useAtsScore hook after 300ms debounce.
    *
    * DO NOT call this function inside a React render or selector.
    * It belongs in a useEffect (via useAtsScore) — NOT in derived state.
    */
   export function computeAtsScore(resume: Resume, keywords: KeywordSet): ScoreResult {
     const resumeText = extractResumeText(resume)

     const hard = scoreCategory(
       keywords.hardRequirements,
       resumeText,
       WEIGHTS.hardRequirements,
       'Hard Requirements'
     )
     const preferred = scoreCategory(
       keywords.preferredSkills,
       resumeText,
       WEIGHTS.preferredSkills,
       'Preferred Skills'
     )
     const tools = scoreCategory(
       keywords.toolsAndTech,
       resumeText,
       WEIGHTS.toolsAndTech,
       'Tools & Tech'
     )
     const soft = scoreCategory(
       keywords.softSkills,
       resumeText,
       WEIGHTS.softSkills,
       'Soft Skills'
     )
     const formatting = checkFormatting(resume)

     const rawTotal =
       hard.weightedScore +
       preferred.weightedScore +
       tools.weightedScore +
       soft.weightedScore +
       formatting.weightedScore

     const totalScore = Math.max(0, Math.min(100, Math.round(rawTotal)))

     // Build matched/missing lists across all keyword categories
     const allKeywords = [
       ...keywords.hardRequirements,
       ...keywords.preferredSkills,
       ...keywords.toolsAndTech,
       ...keywords.softSkills,
     ]

     const allTerms = allKeywords.map((k) => k.term)
     const allResults = matchKeywords(allTerms, resumeText)

     const matchedKeywords = allKeywords
       .filter((_, i) => allResults[i])
       .map((k) => k.term)

     const missingKeywords = allKeywords
       .filter((_, i) => !allResults[i])
       .map((k) => k.term)

     return {
       totalScore,
       breakdown: {
         hardRequirements: hard,
         preferredSkills: preferred,
         toolsAndTech: tools,
         softSkills: soft,
         formatting,
       },
       matchedKeywords,
       missingKeywords,
       computedAt: new Date().toISOString(),
     }
   }
   ```

5. Create `src/hooks/use-ats-score.ts`:

   ```typescript
   'use client'

   import { useEffect, useRef } from 'react'
   import { useAppStore } from '@/lib/store'
   import { computeAtsScore } from '@/lib/scoring/ats-scorer'

   const DEBOUNCE_MS = 300  // Must match DESIGN.md §Two-Stage Architecture "300ms after user stops typing"

   /**
    * Watches resume + keywords in Zustand, debounces 300ms, calls computeAtsScore, dispatches setScore.
    *
    * Mount this hook ONCE in a component that lives for the lifetime of the editor page.
    * It renders nothing — pure side-effect.
    *
    * IMPORTANT: Score computation runs INSIDE useEffect (not during render).
    * DO NOT call computeAtsScore in a selector or derived state — see RESEARCH Pitfall 3.
    *
    * Empty keywords → clearScore() (score panel shows "paste a JD" empty state)
    * No resume → clearScore()
    */
   export function useAtsScore(): void {
     const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null)

     useEffect(() => {
       const unsubscribe = useAppStore.subscribe((state, prevState) => {
         const resumeChanged = state.resume !== prevState.resume
         const keywordsChanged = state.keywords !== prevState.keywords

         if (!resumeChanged && !keywordsChanged) return

         // If keywords or resume just got cleared, clear the score immediately
         if (!state.keywords || !state.resume) {
           if (timerRef.current) {
             clearTimeout(timerRef.current)
             timerRef.current = null
           }
           useAppStore.getState().clearScore()
           return
         }

         // Debounce: cancel any pending computation, schedule a new one in 300ms
         if (timerRef.current) clearTimeout(timerRef.current)
         timerRef.current = setTimeout(() => {
           const { resume, keywords } = useAppStore.getState()
           if (!resume || !keywords) return
           const score = computeAtsScore(resume, keywords)
           useAppStore.getState().setScore(score)
           timerRef.current = null
         }, DEBOUNCE_MS)
       })

       return () => {
         unsubscribe()
         if (timerRef.current) {
           clearTimeout(timerRef.current)
           timerRef.current = null
         }
       }
     }, [])  // Empty deps: subscribe once for the lifetime of the component
   }
   ```
