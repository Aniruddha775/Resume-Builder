   ```tsx
   'use client'

   interface ScoreRingProps {
     score: number  // 0-100
     size?: number  // SVG dimensions in px, default 120
   }

   // Color thresholds: green ≥70, amber 40-69, red <40
   // Note: RESEARCH.md Pattern 7 used ≥80/≥60 thresholds; these are adjusted to match
   // the industry-standard "green=good, amber=warning, red=poor" intuition for ATS scores.
   function getColor(score: number): string {
     if (score >= 70) return '#22c55e'   // Tailwind green-500
     if (score >= 40) return '#f59e0b'   // Tailwind amber-400
     return '#ef4444'                    // Tailwind red-500
   }

   export function ScoreRing({ score, size = 120 }: ScoreRingProps) {
     const strokeWidth = 10
     const radius = (size - strokeWidth * 2) / 2
     const circumference = 2 * Math.PI * radius
     const clampedScore = Math.max(0, Math.min(100, score))
     const dashOffset = circumference * (1 - clampedScore / 100)
     const color = getColor(clampedScore)
     const cx = size / 2
     const cy = size / 2

     return (
       <div
         className="relative flex items-center justify-center"
         style={{ width: size, height: size }}
       >
         <svg
           width={size}
           height={size}
           viewBox={`0 0 ${size} ${size}`}
           role="img"
           aria-label={`ATS Score: ${clampedScore} out of 100`}
         >
           {/* Track circle (background) */}
           <circle
             cx={cx}
             cy={cy}
             r={radius}
             fill="none"
             strokeWidth={strokeWidth}
             className="stroke-muted-foreground/20"
           />
           {/* Progress arc */}
           <circle
             cx={cx}
             cy={cy}
             r={radius}
             fill="none"
             stroke={color}
             strokeWidth={strokeWidth}
             strokeLinecap="round"
             strokeDasharray={circumference}
             strokeDashoffset={dashOffset}
             transform={`rotate(-90 ${cx} ${cy})`}
             style={{ transition: 'stroke-dashoffset 0.4s ease, stroke 0.4s ease' }}
           />
         </svg>
         {/* Score number centered over the SVG */}
         <div className="absolute inset-0 flex flex-col items-center justify-center">
           <span
             className="text-3xl font-bold tabular-nums leading-none"
             style={{ color }}
             aria-hidden="true"
           >
             {clampedScore}
           </span>
           <span className="text-[10px] text-muted-foreground mt-0.5" aria-hidden="true">
             /100
           </span>
         </div>
       </div>
     )
   }
   ```

   ```tsx
   import type { ScoreCategoryBreakdown } from '@/types/score'

   interface CategoryBarProps {
     breakdown: ScoreCategoryBreakdown
   }

   function getBarColor(percentage: number): string {
     if (percentage >= 70) return 'bg-green-500'
     if (percentage >= 40) return 'bg-yellow-400'
     return 'bg-red-400'
   }

   export function CategoryBar({ breakdown }: CategoryBarProps) {
     const { category, matched, total, percentage } = breakdown

     // Don't render empty categories — they contribute 0 points and have no keywords to display
     if (total === 0) return null

     const barColor = getBarColor(percentage)

     return (
       <div className="space-y-1">
         <div className="flex items-center justify-between">
           <span className="text-[11px] text-muted-foreground">{category}</span>
           <span className="text-[11px] font-medium tabular-nums text-foreground">
             {matched}/{total}
           </span>
         </div>
         <div
           className="h-1.5 w-full rounded-full bg-muted overflow-hidden"
           title={`${category}: ${percentage}%`}
         >
           <div
             className={`h-full rounded-full transition-all duration-300 ${barColor}`}
             style={{ width: `${percentage}%` }}
             role="progressbar"
             aria-valuenow={percentage}
             aria-valuemin={0}
             aria-valuemax={100}
             aria-label={`${category}: ${percentage}%`}
           />
         </div>
       </div>
     )
   }
   ```

4. Create `src/components/score-panel/score-panel.tsx`:

   ```tsx
   'use client'

   import { useAppStore } from '@/lib/store'
   import { ScoreRing } from './score-ring'
   import { CategoryBar } from './category-bar'
   import { KeywordChipList } from './keyword-chip-list'

   /**
    * Score Panel — main content for the "Score" tab in the right panel.
    *
    * Reads from Zustand score slice (set by useAtsScore hook).
    * Does NOT compute the score — that's a side effect in useAtsScore.
    * This component is purely presentational: display what's in the store.
    *
    * States:
    * 1. score === null AND keywords === null: Empty state — "Paste a job description"
    * 2. score === null AND keywords !== null: Computing state — keywords set but debounce pending
    * 3. score !== null: Full score display
    */
   export function ScorePanel() {
     const score = useAppStore((s) => s.score)
     const keywords = useAppStore((s) => s.keywords)

     // Empty state — no job description yet
     if (!score && !keywords) {
       return (
         <div className="flex h-full items-center justify-center p-6 text-center">
           <div>
             <p className="text-sm font-medium text-muted-foreground">
               Paste a job description to see your ATS score
             </p>
             <p className="text-[11px] text-muted-foreground/60 mt-1">
               Click "Extract Keywords" in the Job Description panel
             </p>
           </div>
         </div>
       )
     }

     // Keywords set but score not yet computed (first 300ms after extraction)
     if (!score && keywords) {
       return (
         <div className="flex h-full items-center justify-center p-6">
           <p className="text-sm text-muted-foreground" aria-live="polite">
             Computing score…
           </p>
         </div>
       )
     }

     // Score is available — render full panel
     if (!score) return null

     const { totalScore, breakdown, matchedKeywords, missingKeywords } = score
     const categories = [
       breakdown.hardRequirements,
       breakdown.preferredSkills,
       breakdown.toolsAndTech,
       breakdown.softSkills,
     ]

     return (
       <div className="flex flex-col gap-5 p-4 overflow-y-auto">
         {/* Score ring — centered */}
         <div className="flex flex-col items-center gap-1 pt-2">
           <ScoreRing score={totalScore} size={120} />
           <p className="text-[11px] text-muted-foreground mt-1">ATS Score</p>
         </div>

         {/* Category breakdown bars */}
         <div className="space-y-3">
           <p className="text-[11px] font-semibold text-muted-foreground uppercase tracking-wide">
             Breakdown
           </p>
           {categories.map((cat) => (
             <CategoryBar key={cat.category} breakdown={cat} />
           ))}
           <CategoryBar breakdown={breakdown.formatting} />
         </div>

         {/* Matched / Missing keyword chips */}
         <div>
           <p className="text-[11px] font-semibold text-muted-foreground uppercase tracking-wide mb-2">
             Keywords
           </p>
           <KeywordChipList matched={matchedKeywords} missing={missingKeywords} />
         </div>
       </div>
     )
   }
   ```
